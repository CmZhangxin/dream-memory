```bash
#!/bin/bash
# ============================================================
#  Dream Dashboard 一键启动脚本
#  功能：同时启动 Obsidian Vault + Memory Gateway + Next.js Dashboard
#  用法：./start.sh
# ============================================================

set -e

# ─── 配置 ──────────────────────────────────────
VAULT_PATH="/Users/cmzhang/Documents/Obsidian Vault"
DASHBOARD_DIR="$(cd "$(dirname "$0")" && pwd)"
DASHBOARD_PORT=3001
DASHBOARD_URL="http://localhost:${DASHBOARD_PORT}"

# 记忆系统配置
MEMORY_ENABLED="${MEMORY_ENABLED:-true}"
MEMORY_REPO_DIR="${HOME}/.dream-memory/tdai-memory-openclaw-plugin"
MEMORY_REPO_URL="https://github.com/Tencent/TencentDB-Agent-Memory.git"
MEMORY_PORT=8420
MEMORY_DATA_DIR="${HOME}/.dream-memory/data"
MEMORY_LOG_FILE="${HOME}/.dream-memory/gateway.log"
MEMORY_ENV_FILE="${DASHBOARD_DIR}/.env.local"

# ─── 颜色 ──────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC='\033[0m'

echo -e "${CYAN}"
echo "  ╔══════════════════════════════════════╗"
echo "  ║       🧠 Dream Dashboard 启动器       ║"
echo "  ╚══════════════════════════════════════╝"
echo -e "${NC}"

# ─── 检查依赖 ─────────────────────────────────
check_deps() {
  local missing=()
  command -v node >/dev/null 2>&1 || missing+=("node")
  command -v npm >/dev/null 2>&1 || missing+=("npm")

  if [ ${#missing[@]} -gt 0 ]; then
    echo -e "${RED}❌ 缺少依赖: ${missing[*]}${NC}"
    echo -e "${YELLOW}请先安装 Node.js: https://nodejs.org/${NC}"
    exit 1
  fi

  if [ ! -d "$VAULT_PATH" ]; then
    echo -e "${RED}❌ Vault 不存在: $VAULT_PATH${NC}"
    exit 1
  fi

  if [ ! -d "$DASHBOARD_DIR/node_modules" ]; then
    echo -e "${YELLOW}📦 首次运行，正在安装 Dashboard 依赖...${NC}"
    cd "$DASHBOARD_DIR"
    npm install
  fi
}

# ─── 通用清理：按端口杀进程 ─────────────────────
kill_port() {
  local port=$1
  local label=$2
  if command -v lsof >/dev/null 2>&1; then
    local pid=$(lsof -ti :$port 2>/dev/null | head -1)
    if [ -n "$pid" ]; then
      echo -e "${YELLOW}🔄 清理 ${label} 旧进程 (PID: $pid)...${NC}"
      kill "$pid" 2>/dev/null || true
      sleep 1
    fi
  fi
}

cleanup() {
  kill_port "$DASHBOARD_PORT" "Dashboard"
  [ "$MEMORY_ENABLED" = "true" ] && kill_port "$MEMORY_PORT" "Memory Gateway"
}

# ─── 启动 Obsidian ─────────────────────────────
launch_obsidian() {
  if [ -d "/Applications/Obsidian.app" ]; then
    echo -e "${GREEN}✅ 启动 Obsidian...${NC}"
    open -a "Obsidian" "$VAULT_PATH" &
  else
    echo -e "${YELLOW}⚠️  未找到 Obsidian，请手动打开${NC}"
  fi
}

# ─── 准备 Memory 仓库 ─────────────────────────
prepare_memory() {
  [ "$MEMORY_ENABLED" != "true" ] && return

  if [ ! -d "$MEMORY_REPO_DIR" ]; then
    echo -e "${YELLOW}📥 首次运行，克隆 Memory 仓库到 ${MEMORY_REPO_DIR}...${NC}"
    mkdir -p "$(dirname "$MEMORY_REPO_DIR")"
    git clone --depth=1 "$MEMORY_REPO_URL" "$MEMORY_REPO_DIR"
  fi

  if [ ! -d "$MEMORY_REPO_DIR/node_modules" ]; then
    echo -e "${YELLOW}📦 安装 Memory Gateway 依赖（首次较慢，约 2-3 分钟）...${NC}"
    (cd "$MEMORY_REPO_DIR" && npm install --omit=optional)
  fi

  mkdir -p "$MEMORY_DATA_DIR"
}

# ─── 启动 Memory Gateway ──────────────────────
launch_memory() {
  if [ "$MEMORY_ENABLED" != "true" ]; then
    echo -e "${YELLOW}⏭️  跳过 Memory Gateway（MEMORY_ENABLED=false）${NC}"
    return
  fi

  # 加载 .env.local
  if [ -f "$MEMORY_ENV_FILE" ]; then
    set -a
    # shellcheck disable=SC1090
    source "$MEMORY_ENV_FILE"
    set +a
  fi

  if [ -z "$MEMORY_TENCENTDB_LLM_API_KEY" ]; then
    echo -e "${YELLOW}⚠️  未配置 MEMORY_TENCENTDB_LLM_API_KEY${NC}"
    echo -e "${YELLOW}   Memory Gateway 已搭建但不会启动${NC}"
    echo -e "${YELLOW}   配置方法见: dream-dashboard/docs/MEMORY-SETUP.md${NC}"
    echo -e "${YELLOW}   Dashboard 仍会正常启动${NC}"
    return
  fi

  if curl -s "http://localhost:${MEMORY_PORT}/health" >/dev/null 2>&1; then
    echo -e "${GREEN}✅ Memory Gateway 已在运行: ${CYAN}http://localhost:${MEMORY_PORT}${NC}"
    return
  fi

  echo -e "${GREEN}✅ 启动 Memory Gateway (端口: ${MEMORY_PORT})...${NC}"

  (cd "$MEMORY_REPO_DIR" && \
    MEMORY_TENCENTDB_DATA_DIR="$MEMORY_DATA_DIR" \
    node --import tsx src/gateway/server.ts \
    >"$MEMORY_LOG_FILE" 2>&1) &

  echo -n -e "${YELLOW}⏳ 等待 Memory Gateway 就绪"
  for i in $(seq 1 40); do
    if curl -s "http://localhost:${MEMORY_PORT}/health" >/dev/null 2>&1; then
      echo -e "\r${GREEN}✅ Memory Gateway 就绪!${NC}   "
      return
    fi
    echo -n "."
    sleep 0.5
  done

  echo -e "\n${YELLOW}⚠️  Memory Gateway 启动超时${NC}"
  echo -e "${YELLOW}   详见日志: ${MEMORY_LOG_FILE}${NC}"
  echo -e "${YELLOW}   Dashboard 仍会继续启动${NC}"
}

# ─── 启动 Dashboard ────────────────────────────
launch_dashboard() {
  cd "$DASHBOARD_DIR"

  if curl -s "http://localhost:${DASHBOARD_PORT}" >/dev/null 2>&1; then
    echo -e "${GREEN}✅ Dashboard 已在运行: ${CYAN}${DASHBOARD_URL}${NC}"
    return
  fi

  echo -e "${GREEN}✅ 启动 Dashboard (端口: ${DASHBOARD_PORT})...${NC}"
  npm run dev &
  DASH_PID=$!

  echo -n -e "${YELLOW}⏳ 等待 Dashboard 就绪"
  for i in $(seq 1 30); do
    if curl -s "http://localhost:${DASHBOARD_PORT}" >/dev/null 2>&1; then
      echo -e "\r${GREEN}✅ Dashboard 就绪!${NC}   "
      return
    fi
    echo -n "."
    sleep 0.5
    if [ $i -eq 30 ]; then
      echo -e "\n${RED}❌ Dashboard 启动超时${NC}"
      kill $DASH_PID 2>/dev/null || true
      exit 1
    fi
  done
}

# ─── 打开浏览器 ────────────────────────────────
open_browser() {
  echo -e "${CYAN}🌐 打开浏览器...${NC}"
  if command -v open >/dev/null 2>&1; then
    sleep 1
    open "$DASHBOARD_URL"
  elif command -v xdg-open >/dev/null 2>&1; then
    xdg-open "$DASHBOARD_URL"
  fi
}

# ─── 主流程 ────────────────────────────────────
main() {
  check_deps
  cleanup
  launch_obsidian
  prepare_memory
  launch_memory
  launch_dashboard
  open_browser

  echo ""
  echo -e "${GREEN}═══════════════════════════════════════${NC}"
  echo -e "  ${GREEN}🎉 全部启动完成！${NC}"
  echo ""
  echo -e "  Dashboard:      ${CYAN}${DASHBOARD_URL}${NC}"
  if [ "$MEMORY_ENABLED" = "true" ] && [ -n "$MEMORY_TENCENTDB_LLM_API_KEY" ]; then
    echo -e "  Memory Gateway: ${CYAN}http://localhost:${MEMORY_PORT}${NC}"
  fi
  echo -e "  Vault:          ${CYAN}${VAULT_PATH}${NC}"
  echo ""
  echo -e "  ${YELLOW}提示: 按 Ctrl+C 停止所有服务${NC}"
  echo -e "${GREEN}═══════════════════════════════════════${NC}"
  echo ""

  wait
}

trap 'echo -e "\n${YELLOW}👋 正在关闭所有服务...${NC}"; cleanup; exit 0' INT TERM

main "$@"
```

```bash
#!/bin/bash
# ============================================================
#  Dream Dashboard 一键停止脚本
#  功能：停止 Dashboard (3001) + Memory Gateway (8420)
#  用法：./stop.sh
# ============================================================

DASHBOARD_PORT=3001
MEMORY_PORT=8420

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC='\033[0m'

echo -e "${CYAN}"
echo "  ╔══════════════════════════════════════╗"
echo "  ║       🛑 Dream Dashboard 停止器       ║"
echo "  ╚══════════════════════════════════════╝"
echo -e "${NC}"

stop_by_port() {
  local port=$1
  local label=$2
  if ! command -v lsof >/dev/null 2>&1; then
    echo -e "${RED}❌ 缺少 lsof 命令${NC}"
    return 1
  fi

  local pids=$(lsof -ti :"$port" 2>/dev/null)
  if [ -z "$pids" ]; then
    echo -e "${YELLOW}ℹ️  ${label} (端口 ${port}) 未运行${NC}"
    return 0
  fi

  echo -e "${GREEN}🔍 停止 ${label} (端口 ${port}, PID: ${pids})${NC}"
  for pid in $pids; do
    kill "$pid" 2>/dev/null && echo -e "${GREEN}  → SIGTERM PID ${pid}${NC}"
  done

  for i in $(seq 1 10); do
    sleep 0.5
    local still=$(lsof -ti :"$port" 2>/dev/null)
    if [ -z "$still" ]; then
      echo -e "${GREEN}✅ ${label} 已停止${NC}"
      return 0
    fi
  done

  echo -e "${YELLOW}⚠️  ${label} 未在 5 秒内退出，强制结束...${NC}"
  for pid in $(lsof -ti :"$port" 2>/dev/null); do
    kill -9 "$pid" 2>/dev/null && echo -e "${GREEN}  → SIGKILL PID ${pid}${NC}"
  done
}

stop_residue() {
  local next_pids=$(pgrep -f "next dev.*--port ${DASHBOARD_PORT}" 2>/dev/null)
  if [ -n "$next_pids" ]; then
    echo -e "${YELLOW}🧹 清理残留 next dev: ${next_pids}${NC}"
    for pid in $next_pids; do kill "$pid" 2>/dev/null; done
  fi

  local gw_pids=$(pgrep -f "gateway/server.ts" 2>/dev/null)
  if [ -n "$gw_pids" ]; then
    echo -e "${YELLOW}🧹 清理残留 Memory Gateway: ${gw_pids}${NC}"
    for pid in $gw_pids; do kill "$pid" 2>/dev/null; done
  fi
}

main() {
  stop_by_port "$DASHBOARD_PORT" "Dashboard"
  stop_by_port "$MEMORY_PORT"    "Memory Gateway"
  stop_residue

  echo ""
  echo -e "${GREEN}═══════════════════════════════════════${NC}"
  echo -e "  ${GREEN}🎉 全部服务已停止${NC}"
  echo -e "${GREEN}═══════════════════════════════════════${NC}"
  echo ""
}

main "$@"
```
